

public class Program
{
    public static void Main(string[] args)
    {
        Gen<int> collection = new Gen<int>();

        collection.Add(1);
        collection.Add(2);
        collection.Add(3);

        Console.WriteLine("Count: " + collection.Count());
        Console.WriteLine("Element at index 1: " + collection.GetByIndex(1));

        collection.Remove(2);

        Console.WriteLine("Count after removing: " + collection.Count());

        Console.ReadLine();
    }
}

public class Gen<T>
{
    private ArrayList arrayList;
    private List<T> list;
    private Stack<T> stack;
    private Queue<T> queue;

    public Gen()
    {
        arrayList = new ArrayList();
        list = new List<T>();
        stack = new Stack<T>();
        queue = new Queue<T>();
    }

    // Добавление данных в коллекцию
    public void Add(T item)
    {
        arrayList.Add(item);
        list.Add(item);
        stack.Push(item);
        queue.Enqueue(item);
    }

    // Удаление данных из коллекции
    public void Remove(T item)
    {
        arrayList.Remove(item);
        list.Remove(item);
        stack.Pop();
        queue.Dequeue();
    }

    // Получение элемента из коллекции по индексу
    public T GetByIndex(int index)
    {
        return list[index];
    }

    // Получение длины коллекции
    public int Count()
    {
        return list.Count;
    }
}
